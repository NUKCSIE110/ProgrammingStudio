void bubble_sort(int nums[], int length){
    for(int i = 0; i < length-1; i++){
        for(int j = 0; j < length-1; j++){
            if(nums[j] > nums[j+1]){
                temp = nums[j];
                nums[j] = nums[j+1];
                nums[j+1] = temp;
            }
        }
    }
}
