int fun(int arr[],int n);

			printf("Case #%d: It is a B2-Sequence\n",time);
		}

			printf("Case #%d: It is not a B2-Sequence\n",time);
		}
		printf("\n");
	}
	system("PAUSE");
	return 0;
}

int fun(int arr[101],int n){
	int flag = 0;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n;j++){
			if((arr[i]+arr[i+1])==arr[j]){
				flag = 1;
			}
		}
	}
	return flag;
}

#include<stdio.h>
#include<stdlib.h>

int main() {
    int n = 0;
    int counter = 0;
    int flag = 0;
    int times = 1;
    while (scanf("%d", &n) != EOF) {
        int *record = (int*)malloc(sizeof(int) * ((1 + n)*n / 2));
        int *nums = (int*)malloc(sizeof(int) * n);
        counter = 0;
        flag = 0;
        for (int i = 0; i < n; i++) {
            scanf("%d", &nums[i]);
        }
        for (int i = n - 1; i >= 0; i--) {
            for (int j = i; j >= 0; j--) {
                for (int k = 0; k < counter; k++) {
                    if (nums[i] + nums[j] == record[k] || nums[j] > nums[i]) {
                        flag = 1;
                        break;
                    }
                }
                if (flag) break;
                record[counter] = nums[i] + nums[j];
                counter++;
            }
            if (flag) break;
        }

        if (flag) {
            printf("Case #%d: It is not a B2-Sequence.\n\n", times);
        }
        else {
            printf("Case #%d: It is a B2-Sequence.\n\n", times);
        }
        times++;
        
    }
    return 0;
}

#include<stdio.h>
#include<stdlib.h>

int main()
{
	int n,count = 1;
	
	while(scanf("%d",&n) != EOF){
		int *arr = (int*)malloc(n*sizeof(int));
		int sum = 0,error = 0,k = 0;
		
		for(int i = 1;i <= n;i++)
			sum += i;
		int *ans = (int*)malloc(sum*sizeof(int));
		
		for(int i = 0;i < n;i++){
			scanf("%d",&arr[i]);
			if(arr[0] == 0)
				error++;
			if(i >= 1)
				if(arr[i] < arr[i-1])
					error++;
		}
		
		if(error == 0){
			for(int i = 0;i < n;i++)
				for(int j = i;j < n;j++){
					ans[k] = arr[i] + arr[j];
					k++;
				}
		
			for(int i = 0;i < k;i++){
				for(int j = 0;j < k;j++){
					if(i == j)
						continue;
					else if(ans[i] == ans[j]){
						error++;
						break;
					}
				}
				if(error != 0)
				break;
			}
		}
		if(error != 0)
		printf("Case #%d: It is not a B2-Sequence.\n",count);
		else
		printf("Case #%d: It is a B2-Sequence.\n",count);
		
		count++;
		free(arr);
		free(ans);
		printf("\n");
	}
	
	return 0;
}
