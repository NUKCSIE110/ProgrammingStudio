#include <stdio.h>
int main()
{
	double x[4]={0};
	double y[4]={0};
	double mx=0,my=0;
	double ax=0,ay=0;
	while (scanf("%lf %lf %lf %lf %lf %lf %lf %lf",&x[0],&y[0],&x[1],&y[1],&x[2],&y[2],&x[3],&y[3])!=EOF)
	{
		if(x[0]==x[1]&&y[0]==y[1])
		{
			mx=(x[2]+x[3])/2;
			my=(y[2]+y[3])/2;
			ax=mx-(x[0]-mx);
			ay=my-(y[0]-my);
		}
		else if(x[0]==x[2]&&y[0]==y[2])
		{
			mx=(x[1]+x[3])/2;
			my=(y[1]+y[3])/2;
			ax=mx-(x[0]-mx);
			ay=my-(y[0]-my);
		}
		else if(x[0]==x[3]&&y[0]==y[3])
		{
			mx=(x[2]+x[1])/2;
			my=(y[2]+y[1])/2;
			ax=mx-(x[0]-mx);
			ay=my-(y[0]-my);
		}
		else if(x[1]==x[2]&&y[1]==y[2])
		{
			mx=(x[0]+x[3])/2;
			my=(y[0]+y[3])/2;
			ax=mx-(x[1]-mx);
			ay=my-(y[1]-my);
		}
		else if(x[1]==x[3]&&y[1]==y[3])
		{
