## 14:公式解
``` c++
#include<stdio.h>

int main() {
    int n = 0;
    int ax, ay, bx, by;
    int sum_a = 0, sum_b = 0;
    scanf("%d", &n);
    for (int k = 1; k <= n;k++) {
        scanf("%d %d %d %d", &ax, &ay, &bx, &by);
        sum_a = ax + ay;
        sum_b = bx + by;
        //同一層時
        if (sum_a == sum_b) {
            printf("Case %d: %d\n", k, ax - ay);
            continue;
        }
        //不同層時
        int diff = sum_b - sum_a;
        int diffSum = 0;
        for (int i = sum_a; i < sum_b; i++) {
            diffSum += i;
        }
        diffSum += diff;
        int ans = diffSum - ax + bx;
        printf("Case %d: %d\n", k, ans);
        sum_a = 0;
        sum_b = 0;
    }
    return 0;
}
```

```c++
int ans(int x,int y)
{
	if (x==0&&y==0)
	{
		return 0;
	}
	else if(x==0&&y==1)
	{
		return 1;
	}
	else if(x==1&&y==0)
	{
		return 2;
	}
	else if(x==0)
	{
		return 1+ans(x+y-1,0);	
	}
	else if(y==0)
	{
		return 1+ans(1,x-1);
	}
	else if(y==1)
	{
		return 1+ans(0,x+y);
	}
	else 
	{
		return 1+(x+1,y-1);
	}
}

	{

		{
			A=ans(a,b)-ans(c,d);
```

## 41:一步一步走的解法
``` c++
#include <stdio.h>

int count(int x0,int y0,int x,int y)
{
	int steps=0;
	while(1)
	{
		if (x0==x && y0==y) break;
		if(x0==0 && y0==0){
			y0=y0+1;
			steps++;
		}
		else if(y0==0){
			y0=x0+1;
			x0=0;
			steps++;
		}
		else {
			y0=y0-1;
			x0=x0+1;
			steps++;
		}
	}
	return steps;
}
int main()
{
	int n=0;
	int x0=0,y0=0;
	int x=0,y=0;

	while(scanf("%d",&n) !=EOF){
		if(n==0) break;
		for(int i=0;i<n;i++){
		scanf("%d %d %d %d",&x0,&y0,&x,&y);
		printf("Case %d: %d\n",i+1,count(x0,y0,x,y));
	}
	}
	return 0;
}
```
