``` c++
#include<stdio.h>

void banglaNumber(long long n) {
	if (n == 0) return;
    
	if (n / 10000000) {
		banglaNumber(n / 10000000);
		printf(" kuti");
		n %= 10000000;
	}

	if (n / 100000) {
		banglaNumber(n / 100000);
		printf(" lakh");
		n %= 100000;
	}

	if (n / 1000) {
		banglaNumber(n / 1000);
		printf(" hajar");
		n %= 1000;
	}

	if (n / 100) {
		banglaNumber(n / 100);
		printf(" shata");
		n %= 100;
	}

	if (n) printf(" %lld", n);
}

int main() {
	long long n;
	int num = 1;
	while (scanf("%lld", &n) != EOF) {
		printf("%4d.", num);
		if (n) 
			banglaNumber(n);
		else 
			printf(" 0");
		printf("\n");
		num++;
	}
	return 0;
}
```
