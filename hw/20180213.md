## 14:過年懶得寫 加了一點點的黑魔法
``` c++
#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
#include<vector>

using namespace std;

int main() {
    int n;
    string str;
    vector<pair<string, int>> list;
    int idx;
    scanf("%d", &n);
    getchar();
    for(int count = 0;count < n;count++) {
        getline(cin, str);
        size_t i = str.find(" ");
        string country = str.substr(0, i);
        //cout << country << endl;
        int exist = 0;
        for (idx = 0; idx < list.size(); idx++) {
            if (country == list[idx].first) {
                exist = 1;
                list[idx].second++;
                break;
            }
        }
        if (!exist) list.push_back(make_pair(country, 1));
    }
    sort(list.begin(), list.end());
    for (idx = 0; idx < list.size(); idx++) {
        cout << list[idx].first << " " << list[idx].second << endl;
    }
    return 0;
}
```
