## 9:

#include<stdio.h>
int count(int a,int b){
	int sum=0;
	if(a%2==0)a++;
	for (int i=a;i<=b;i+=2){
		sum+=i;
	}
	return sum;
}
int main(){
	int num,a,b;
	scanf("%d",&num);
	for(int i=1;i<=num;i++){
		scanf("%d",&a);
		scanf("%d",&b);
		printf("Case %d: %d\n",i,count(a,b));
	}
	return 0;
}

#include<stdio.h>

int main()
{
	int a;

	int time = 1;

	scanf("%d",&n);
	if(n < 1 || n > 100)
	return 0;
	for(int i = 0;i < n;i++)
	{
		scanf("%d %d",&a,&b);
		if(a > b)
			return 0;
		else if(a > 100 || a < 0 || b > 100 || b < 0)
			return 0;
	
		if(a % 2 == 0 && b % 2 == 0)
			for(int i = 1;a + i <= b - 1;i += 2)
				sum += a+i;
		else if(a % 2 != 0 && b % 2 == 0)
			for(int i = 0;a + i <= b - 1;i += 2)
				sum += a+i;
		else if(a % 2 == 0 && b % 2 != 0)
			for(int i = 1;a + i <= b;i += 2)
				sum += a+i;
		else if(a % 2 != 0 && b % 2 != 0)
			for(int i = 0;a + i <= b;i += 2)
				sum += a+i;
		printf("Case %d: %d\n",time,sum);
		time++;
		sum = 0;
    }    return 0;
