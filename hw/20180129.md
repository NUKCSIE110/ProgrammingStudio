``` c++
#include<stdio.h>
#include<stdlib.h>

int main()
{
    long long member, day;

	while(scanf(" %lld %lld", &member, &day) != EOF)
	{
		for(int i = member; i > 0; i++)
		{
			day -= i;
			if(day <= 0)
			{
				printf("%d\n",i);
				break;	 
			}
		}
	}
		return 0;
}
```

## 14:
``` c++
#include<stdio.h>

int main() {
    long long int s = 0;
    long long int d = 0;
    while (scanf_s("%lld %lld", &s, &d) != EOF) {
        while (d - s > 0) {
            d -= s;
            s++;
        }
        printf("%lld\n", s);
    }
    return 0;
}
```

``` c++
#include<stdio.h>
#include<stdlib.h>
#pragma warning (disable:4996)

int main(void)
{
	long long int S = 0;
	long long int D = 0;
	long long int count = 0;
	long long int sum = 0;
	while (scanf("%lld %lld", &S, &D) != EOF)
	{
		while (sum< D) {
			count++;
			sum += S + count;
			
		}
		printf("%lld\n", count+S);
		sum = 0;
		count = 0;
	}
	/*system("pause");*/
	return 0;
}
```

``` c++
#include<stdio.h>

int main()
{
	long long int day,size,n,ans;
	while(scanf("%lld %lld",&size,&day) != EOF){
		n = size;
		ans = day;
	
		while(1){
			ans -= n;
			if(ans > 0)
				n++;
			else if(ans <= 0)
				break;
		}
		printf("%lld\n",n);
	}
}
```
