``` c++
#include<stdio.h>
#include<stdlib.h>

int main()
{
    long long member, day;

	while(scanf(" %lld %lld", &member, &day) != EOF)
	{
		for(int i = member; i > 0; i++)
		{
			day -= i;
			if(day <= 0)
			{
				printf("%d\n",i);
				break;	 
			}
		}
	}
		return 0;
}
```

## 14:
``` c++
#include<stdio.h>

int main() {
    long long int s = 0;
    long long int d = 0;
    while (scanf_s("%lld %lld", &s, &d) != EOF) {
        while (d - s > 0) {
            d -= s;
            s++;
        }
        printf("%lld\n", s);
    }
    return 0;
}
```
