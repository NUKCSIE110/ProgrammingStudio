## 9:迴圈OP
``` c++
#include<stdio.h> 
int GCD(int a, int b) {
    int max = 0;
    if (a >= b) {
        for (int i = 1; i <= b; i++) {
            if (a%i == 0 && b%i == 0) {
                max = i;
            }
        }
    }
    else {
        for (int i = 1; i <= a; i++) {
            if (a%i == 0 && b%i == 0) {
                max = i;
            }
        }
    }
    return max;
}
int main() {
    while (1) {
        int N;
        scanf("%d", &N);
        if (N == 0) break;
        int G = 0;
        for (int i = 1; i<N; i++)
            for (int j = i + 1; j <= N; j++) {
                G += GCD(i, j);
            }
        printf("%d\n", G);
    }
    return 0;
}
```

## 14:輾轉相除法求最大公因數
``` c++
#include<stdio.h>

int gcd(int a, int b) {
    if (b) while ((a %= b) && (b %= a));
    return a + b;
}

int main() {
    int n = 0;
    while (scanf("%d", &n)) {
        int G = 0;
        if (n == 0) break;
        else {
            for (int i = 1; i<n; i++) {
                for (int j = i + 1; j <= n; j++) {
                    G += gcd(i, j);
                }
            }
        }
        printf("%d\n", G);
    }
    return 0;
}
```

## 39:輾轉相除法
``` c++
#include<stdio.h>

int GCD(int i,int j)
{
	if(i == 0 || j == 0)
		return 0;
	else if(i > j){
		i -= j;
		GCD(i,j);
	}
	else if(j > i){
		j -= i;
		GCD(i,j);
	}
	else
		return i;
}

int main()
{
	int n[101];
	int g = 0;
	
	for(int i = 0;i < 101;i++){
		scanf("%d",&n[i]);
		//printf("1\n");
		if(n[i] == 0)
		break;
		int z;
		z = n[i];
	
		for(int j = 0;j < z;j++){
			for(int k = (j + 1);k <= z;k++){
				g += GCD(j,k);
			}
		}
		printf("%d\n",g);
		g = 0;
	}
	return 0;
}
```
