```c++
#include <cstdio>
#include <cstring>
#include <stdlib.h>
#include <cctype>
int main(){
    char sentence[102][102] = { 0 };
    int longest = 0;
    int n = 0;
    while (fgets(sentence[n], 102, stdin) != NULL) {
        if (strlen(sentence[n]) > longest) {
            longest = strlen(sentence[n]);
        }
        n++;
    }
    for (int i = 0; i < longest-1; i++) {
        for (int j = n-1; j >=0; j--) {
            if (iswprint(sentence[j][i]) && sentence[j][i] != '\t')
                printf("%c", sentence[j][i]);
            else
                printf(" ");
        }
        printf("\n");
    }
    return 0;
}
```

```c++
#include "stdio.h"
#include "stdlib.h"

int main()
{
	char tmp;
	int max = 0;
	char matrix[150][150] = { ' ' };
	int row=0, col=0;
	while (tmp = getchar()) {
		if (tmp == '\n') {
			row++;
			col = 0;
		}
		else if (tmp == EOF) {
			col--;
			row--;
			break;
		}
		else if (tmp == '\0') {
			matrix[row][col] = ' ';
			if (col > max) max = col;
			col++;
		}
		else {
			matrix[row][col] = tmp;
			if (col > max) max = col;
			col++;
		}
	}
	for (int i = 0; i <= max; i++) {
		for (int j = row; j >= 0; j--) {
			printf("%c", matrix[j][i] != '\0' ? matrix[j][i] : ' ');
			if (j == 0) puts("");
		}
	}
    return 0;
}
```
