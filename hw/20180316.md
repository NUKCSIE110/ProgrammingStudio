#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<ctype.h>
#pragma warning(disable:4996)

int main()
{
	int n,c=0,max=0; int count[27] = {0};
	scanf("%d", &n);
	while (c != n+1)
	{
		char a[1000] = {}; int b[1000] = {};
		fgets(a, 1000, stdin);
	for (int i = 0; i < strlen(a); i++)
			b[i] = a[i] - 'a';
		for (int i = 0; i < strlen(a); i++)
		{
			for (int j = 0; j < 26; j++)
			{
				if (b[i] == j || b[i] == j + 'A' - 'a')
					count[j]++;
					else continue;
			}
		}
	c++;
	memset(a, 0, sizeof(a)); memset(b, 0, sizeof(b));
	}
	for (int i = 0; i < 26; i++)
	{
		if (count[i] > max)
			max = count[i];
	}
	for (int j = max; j > 0; j--)
	{
		for (int i = 0; i < 26; i++)
		{
			if (count[i] == 0)  continue;
			else if (count[i] == j) printf("%c %d\n", (i)+'A', count[i]);
		}
	}
	return 0;
}
