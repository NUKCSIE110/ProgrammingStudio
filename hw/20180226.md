``` c++
#include <stdio.h>
#include <stdlib.h>

int main()
{
	int input = 0,digit[10] = {0},output = 0;
	while(1)
	{
		scanf("%d",&input);
		if(input == 0) break;
		output = 0;
		while(1)
		{
			for(int i = 0;i < 10;i++)
			{
				digit[i] = input % 10;
				input /= 10;
				output += digit[i];
			}
			if(output >= 10)
				input = output;
			else break;
			output = 0;
		}
		printf("%d\n",output);
	}
	return 0;
}
```

``` c++
#include<stdio.h>
#include<math.h>
#include<stdlib.h>
int digitSum(int n) {
    int sum = 0;
    while (n) {
        sum += n % 10;
        n /= 10;
    }
    return sum;
}

int main() {
    long long int num;
    while (scanf("%d", &num)) {
        if (num == 0) break;
        while (num > 10) {
            num = digitSum(num);
        }
        printf("%d\n", num);
    }
    return 0;
}
```
