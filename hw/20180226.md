## 14:
``` c++
#include<stdio.h>
#include<math.h>
#include<stdlib.h>
int digitSum(int n) {
    int sum = 0;
    while (n) {
        sum += n % 10;
        n /= 10;
    }
    return sum;
}

int main() {
    long long int num;
    while (scanf("%d", &num)) {
        if (num == 0) break;
        while (num > 10) {
            num = digitSum(num);
        }
        printf("%d\n", num);
    }
    return 0;
}
```
