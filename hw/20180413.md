#include<stdio.h>
#include<string.h>

int min(int a, int b) {
    if (a >= b) return b;
    else return a;
}

int main() {
    int counter1[26] = { 0 };
    int counter2[26] = { 0 };
    char str1[1002];
    char str2[1002];
    while(fgets(str1, 1002,stdin) != NULL){
        fgets(str2, 1002, stdin);
        if (strlen(str1) == 1 && str1[0] == '\n' || strlen(str2) == 1 && str2[0] == '\n') {
            continue;
        }
        for (int i = 0; i < strlen(str1)-1; i++) {
            counter1[str1[i] - 'a']++;
        }
        for (int i = 0; i < strlen(str2) - 1; i++) {
            counter2[str2[i] - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if (counter1[i] && counter2[i]) {
                for (int j = 0; j < min(counter1[i], counter2[i]); j++) {
                    printf("%c", ('a' + i));
                }
            }
        }
        printf("\n");
        for (int i = 0; i < 26; i++) {
            counter1[i] = 0;
            counter2[i] = 0;
        }
    }
    return 0;
}

	int a_count[26] = { 0 };
	int min;
	while (fgets(a,1500,stdin))
	{
		fgets(b, 1500, stdin);
		for (int i = 0; i < 26; i++)
		{
			for (int j = 0; j < strlen(a); j++)
			{
				if (a[j] == i + 'a')
					a_count[i]++;
			}
		}
			for (int i = 0; i < strlen(a); i++)
				{
					for (int j = 0; j < strlen(b); j++)
					{
						if (a[i] == b[j])
						{
							c[a[i] - 'a']++;
						}
					}
				}
				for (int i = 0; i < 26; i++)
				{
					if (a_count[i] != 0)
					{
						if (a_count[i] < c[i] / a_count[i]) min = a_count[i];
						else min = c[i] / a_count[i];
						for (int j = 0; j < min; j++)
							printf("%c", i + 'a');
					}
				}
		printf("\n");
		memset(a, 0, sizeof(a));
		memset(b, 0, sizeof(b));
