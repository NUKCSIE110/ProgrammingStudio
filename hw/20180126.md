```c++
		int count[3]={n,n,n};
		int n_use[3]={n,n+1,n+2};
		int max=0;
		for(int i=0;i<3;i++){
			while(n_use[i]>=3){
				count[i]=count[i]+(n_use[i]/3);
				n_use[i]=(n_use[i]/3)+(n_use[i]%3);
			}
			if(n_use[i]>=i){
				if(count[i]>max){
					max=count[i];
				}
			}
		}
		printf("%d\n",max);
```

``` c++
#include<stdio.h>

int main()
{
	int k;
	while(scanf("%d",&k) != EOF){
		int sum = 0;
		
		while(1){
			if(k == 1){
				sum += 1;
				break;
			}
			
			else if(k == 2){
				sum += 2;
				k = 1;
			}
			
			else if(k % 3 == 0){
				sum += k;
				k /= 3;
			}
			
			else if(k % 3 == 1){
				sum += k;
				k /= 3;
				k += 1;
				sum -= 1;
			}
			
			else if(k % 3 == 2){
				sum += k;
				k /= 3;
				k += 2;
				sum -= 2;
			}
		}
		printf("%d\n",sum);
	}
}
```
