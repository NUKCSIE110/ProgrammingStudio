## 14: 借瓶子不科學._.
``` c++
#include<stdio.h>

int main() {
    int n = 0;
    int cola = 0;
    while (scanf_s("%d", &n) != EOF) {
        cola = n;
            if (n == 1) {
                printf("1\n");
            }
            else {
                while (n != 1) {
                    if (n == 2) n++;
                    cola += n / 3;
                    printf("%d %d\n", n / 3, n % 3);
                    n = n / 3 + n % 3;
                    
                }
                printf("%d\n", cola);
            }
        
    }

    return 0;
}
```
